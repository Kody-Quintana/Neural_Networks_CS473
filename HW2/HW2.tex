\question
Given:

$\vec{X} = [1.5\quad 3.6\quad 7.0\quad 10.0]^T
$

$\vec{W} = [0.1\quad 0.3\quad 0.5\quad 0.7]^T
$

find:

\begin{enumerate}[(a)]
\item $h = \vec{W}^T \vec{X} $

\item $z = \frac
	{1}
	{1 + e^{-h}}$

\item will the nueron fire?
\end{enumerate}
\closequestion

\begin{enumerate}[(a)]
\item $h = \vec{W}^T \vec{X} = 11.73$

\item $z = \frac
	{1}
	{1 + e^{-h}} = 
	\frac
	{1}
	{1 + e^{-11.73}} =
	\frac
	{1}
	{1 + 0.00000804869} =
	0.999992
$

\item Yes, because $0.999992 \geq 0.5$
\end{enumerate}
